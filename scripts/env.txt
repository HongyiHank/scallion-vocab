# Container build environment for scallion-vocab
# Source this file in scripts to keep config in one place.
#
# Usage:  source "$(dirname "$0")/env.txt"
#
# jdk_version:         Java major version. Set to "latest" to auto-resolve the newest
#                      OpenJDK available in apt during container build.
#                      Examples: 17, 21, latest
# platform:            Container runtime (podman/docker)
# container_location:  Image name or path (e.g. localhost/voc-builder)
# container_name:      Daemon container name (for warm Gradle daemon)
# android_platform:    Android SDK platform version (must match compileSdk in build.gradle.kts).
#                      Set to "latest" to auto-resolve the newest via sdkmanager.
#                      Examples: 34, 35, latest
# storage_dir:         Container image storage path (empty = default = ~/.local/share/containers/).
#                      Only needed if you want images on another drive.
#                      On Windows + Podman Desktop, leave empty (uses VM's ext4).
#                      On Linux, if target filesystem isn't native Linux (e.g., NTFS mount),
#                      create a btrfs disk image on it and loop-mount it, then point here.
#                      Example: storage_dir=/mnt/containers
# keystore_file:        Full path to the release keystore.
#                       Can be any path on the host; the script bind-mounts
#                       its parent directory into the container automatically.
#                       Key alias is auto-detected from the keystore file.
#                       Examples:
#                         keystore_file=/mnt/intel/Data/Keystore/scallion-vocab-release.keystore
#                         keystore_file=${HOME}/.android/debug.keystore
# keystore_pass:        Keystore password (single-quoted to preserve special chars).
# app_package_name:     Android application ID (package name).
#                       Must match Dioxus.toml [bundle].identifier.
#                       Changing this changes the APK package name.
#
# When changing jdk_version or android_platform, rebuild the container image.
# Containerfile lives at the project root (auto-discovered).

jdk_version=latest
platform=podman
container_location=localhost/voc-builder
container_name=voc-builder-daemon
android_platform=latest
storage_dir=
keystore_file=
keystore_pass=''
app_package_name=com.scallion.vocab
